```typescript
class AIWritingEngine {
  constructor(
    private aiService: AIService,
    private consistencyEngine: ConsistencyEngine,
    private promptManager: PromptManager,
    private feedbackProcessor: FeedbackProcessor,
    private versionManager: VersionManager
  ) {}

  // 生成设定
  async generateSetup(config: SetupGenerationConfig): Promise<Setup> {
    const prompt = this.promptManager.getPrompt('generate_setup', {
      projectName: config.projectName,
      genre: config.genre,
      description: config.description,
      targetWordCount: config.targetWordCount,
    });

    const response = await this.aiService.complete(prompt, {
      temperature: 0.7,
      maxTokens: 4000,
    });

    const setup = this.parseSetupResponse(response);

    // 创建初始版本
    await this.versionManager.createVersion(
      config.projectId,
      'SETUP',
      setup.id,
      JSON.stringify(setup, null, 2),
      'Initial setup generation',
      'ai_system'
    );

    return setup;
  }

  // 生成大纲
  async generateOutline(config: OutlineGenerationConfig): Promise<Outline> {
    const setup = await this.storage.getSetup(config.projectId);

    const prompt = this.promptManager.getPrompt('generate_outline', {
      setup,
      targetChapters: config.targetChapters,
      targetWordCount: config.targetWordCount,
    });

    const response = await this.aiService.complete(prompt, {
      temperature: 0.7,
      maxTokens: 8000,
    });

    const outline = this.parseOutlineResponse(response);

    // 生成拓扑图
    await this.consistencyEngine.buildTopology(config.projectId, outline);

    // 创建初始版本
    await this.versionManager.createVersion(
      config.projectId,
      'OUTLINE',
      outline.id,
      JSON.stringify(outline, null, 2),
      'Initial outline generation',
      'ai_system'
    );

    return outline;
  }

  // 生成章节内容
  async generateContent(config: ContentGenerationConfig): Promise<ChapterContent> {
    const { projectId, chapterIndex } = config;

    // 获取上下文
    const context = await this.buildChapterContext(projectId, chapterIndex);

    // 一致性检查前置
    const consistencyIssues = await this.consistencyEngine.preCheck(projectId, chapterIndex);
    if (consistencyIssues.length > 0) {
      console.warn('Consistency issues detected:', consistencyIssues);
    }

    // 构建 Prompt
    const prompt = await this.buildChapterPrompt(context, consistencyIssues);

    // 生成内容
    const response = await this.aiService.complete(prompt, {
      temperature: 0.8,
      maxTokens: 6000,
    });

    const content = this.parseContentResponse(response);

    // 一致性检查后
    const postCheckIssues = await this.consistencyEngine.postCheck(
      projectId,
      chapterIndex,
      content
    );

    if (postCheckIssues.length > 0) {
      content.consistencyChecks = postCheckIssues;
    }

    // 更新拓扑图
    await this.consistencyEngine.updateTopology(projectId, chapterIndex, content);

    // 创建初始版本
    await this.versionManager.createVersion(
      projectId,
      'CHAPTER',
      content.id,
      content.content,
      `Initial generation of chapter ${chapterIndex}`,
      'ai_system'
    );

    return content;
  }

  // 构建章节上下文
  private async buildChapterContext(
    projectId: string,
    chapterIndex: number
  ): Promise<ChapterContext> {
    const project = await this.storage.getProject(projectId);
    const setup = await this.storage.getSetup(projectId);
    const outline = await this.storage.getOutline(projectId);
    const chapterOutline = outline.chapters[chapterIndex - 1];

    // 获取前文摘要（最近3章）
    const previousChapters = await this.storage.getChapters(projectId, {
      start: Math.max(1, chapterIndex - 3),
      end: chapterIndex - 1,
    });

    const previousSummaries = previousChapters.map((c) => ({
      index: c.chapterIndex,
      title: c.title,
      summary: this.generateSummary(c.content),
    }));

    // 获取相关角色信息
    const characterIds = chapterOutline.characters;
    const characters = await Promise.all(
      characterIds.map((id) => this.storage.getCharacter(projectId, id))
    );

    // 获取待回收伏笔
    const pendingForeshadowing = await this.consistencyEngine.getPendingForeshadowing(
      projectId,
      chapterIndex
    );

    return {
      project,
      setup,
      outline,
      chapterOutline,
      previousSummaries,
      characters,
      pendingForeshadowing,
    };
  }

  // 构建章节 Prompt
  private async buildChapterPrompt(
    context: ChapterContext,
    consistencyIssues: ConsistencyIssue[]
  ): Promise<string> {
    const sections: string[] = [];

    sections.push(this.promptManager.getSystemPrompt('novel_writing'));

    sections.push(`
【世界观设定】
${context.setup.worldBuilding.background}

【核心规则】
${context.setup.worldBuilding.rules}
`);

    sections.push(`
【本章出场角色】
${context.characters
  .map(
    (c) => `
- ${c.name}: ${c.personality}
  背景: ${c.background}
  当前动机: ${c.motivation}
`
  )
  .join('\n')}
`);

    if (context.previousSummaries.length > 0) {
      sections.push(`
【前文摘要】
${context.previousSummaries.map((s) => `第${s.index}章 ${s.title}: ${s.summary}`).join('\n')}
`);
    }

    sections.push(`
【本章大纲】
标题: ${context.chapterOutline.title}
摘要: ${context.chapterOutline.summary}
目的: ${context.chapterOutline.purpose}
情感弧线: ${context.chapterOutline.emotionalArc}

场景列表:
${context.chapterOutline.scenes.map((s, i) => `${i + 1}. ${s.setting} - ${s.purpose}`).join('\n')}
`);

    if (context.pendingForeshadowing.length > 0) {
      sections.push(`
【待回收伏笔】
${context.pendingForeshadowing.map((f) => `- ${f.description} (重要性: ${f.importance})`).join('\n')}
`);
    }

    if (consistencyIssues.length > 0) {
      sections.push(`
【一致性注意事项】
${consistencyIssues.map((i) => `- ${i.description}`).join('\n')}
`);
    }

    sections.push(`
【输出要求】
- 字数: ${context.chapterOutline.wordCount}字左右
- 风格: ${context.project.metadata.style}
- 使用标准中文标点
- 段落清晰，对话单独成行
`);

    return sections.join('\n\n---\n\n');
  }
}
```

```typescript
  getPrompt(templateId: string, variables: Record<string, string>): string {
    const template = this.templates.get(templateId);
    if (!template) throw new Error(`Template not found: ${templateId}`);

    let prompt = template.template;
    for (const [key, value] of Object.entries(variables)) {
      prompt = prompt.replace(new RegExp(`{{${key}}}`, 'g'), value);
    }

    if (this.userPreferences) {
      prompt = this.applyPreferences(prompt);
    }

    return prompt;
  }

  getSystemPrompt(type: string): string {
    return this.templates.get(`system_${type}`)?.template || '';
  }

  private applyPreferences(prompt: string): string {
    if (!this.userPreferences) return prompt;

    const prefs = this.userPreferences.stylePreferences;
    const additions: string[] = [];

    if (prefs.descriptionDensity > 0.5) {
      additions.push('- 增加环境描写和细节刻画');
    } else if (prefs.descriptionDensity < -0.5) {
      additions.push('- 保持简洁，减少环境描写');
    }

    if (prefs.dialogueRatio > 0.5) {
      additions.push('- 增加对话比例，通过对话推进情节');
    }

    if (prefs.pacingSpeed > 0.5) {
      additions.push('- 加快节奏，减少铺垫');
    } else if (prefs.pacingSpeed < -0.5) {
      additions.push('- 放缓节奏，增加细节和铺垫');
    }

    if (additions.length > 0) {
      prompt += '\n\n【用户风格偏好】\n' + additions.join('\n');
    }

    return prompt;
  }

  setUserPreferences(preferences: UserPreferenceProfile): void {
    this.userPreferences = preferences;
  }
}
```

```typescript
class TokenBudgetManager {
  private readonly DEFAULT_BUDGET: TokenBudget = {
    total: 6000,
    systemPrompt: 800,
    characters: 1000,
    foreshadows: 600,
    previousChapters: 1200,
    worldFacts: 800,
    outputFormat: 400,
    reserved: 1200,
  };

  constructor(private baseBudget: TokenBudget = DEFAULT_BUDGET) {}

  allocate(context: AllocationContext): AllocatedBudget {
    const { contentLength, chapterPosition, complexity } = context;

    const lengthFactor = this.calculateLengthFactor(contentLength);
    const positionFactor = this.calculatePositionFactor(chapterPosition);
    const complexityFactor = this.calculateComplexityFactor(complexity);

    const adjustedBudget: TokenBudget = {
      total: this.baseBudget.total,
      systemPrompt: this.baseBudget.systemPrompt,
      characters: Math.floor(this.baseBudget.characters * lengthFactor.characters),
      foreshadows: Math.floor(this.baseBudget.foreshadows * complexityFactor.foreshadows),
      previousChapters: Math.floor(this.baseBudget.previousChapters * positionFactor.previous),
      worldFacts: this.baseBudget.worldFacts,
      outputFormat: this.baseBudget.outputFormat,
      reserved: this.baseBudget.reserved,
    };

    const sum =
      adjustedBudget.systemPrompt +
      adjustedBudget.characters +
      adjustedBudget.foreshadows +
      adjustedBudget.previousChapters +
      adjustedBudget.worldFacts +
      adjustedBudget.outputFormat +
      adjustedBudget.reserved;

    if (sum > adjustedBudget.total) {
      const ratio = adjustedBudget.total / sum;
      adjustedBudget.characters = Math.floor(adjustedBudget.characters * ratio);
      adjustedBudget.foreshadows = Math.floor(adjustedBudget.foreshadows * ratio);
      adjustedBudget.previousChapters = Math.floor(adjustedBudget.previousChapters * ratio);
      adjustedBudget.worldFacts = Math.floor(adjustedBudget.worldFacts * ratio);
    }

    return {
      budget: adjustedBudget,
      factors: {
        length: lengthFactor,
        position: positionFactor,
        complexity: complexityFactor,
      },
    };
  }

  private calculateLengthFactor(contentLength: number): BudgetFactors {
    if (contentLength > 5000) {
      return { characters: 1.3, foreshadows: 1.2, previous: 1.1 };
    } else if (contentLength > 3000) {
      return { characters: 1.1, foreshadows: 1.1, previous: 1.0 };
    }
    return { characters: 1.0, foreshadows: 1.0, previous: 0.9 };
  }

  private calculatePositionFactor(chapterPosition: number): BudgetFactors {
    if (chapterPosition <= 3) {
      return { characters: 1.2, foreshadows: 0.9, previous: 0.8 };
    } else if (chapterPosition >= 20) {
      return { characters: 1.0, foreshadows: 1.2, previous: 1.3 };
    }
    return { characters: 1.0, foreshadows: 1.0, previous: 1.0 };
  }

  private calculateComplexityFactor(complexity: number): BudgetFactors {
    if (complexity >= 4) {
      return { characters: 1.2, foreshadows: 1.4, previous: 1.1 };
    } else if (complexity >= 2) {
      return { characters: 1.1, foreshadows: 1.2, previous: 1.0 };
    }
    return { characters: 1.0, foreshadows: 1.0, previous: 1.0 };
  }

  estimateTokens(text: string): number {
    const chineseChars = (text.match(/[\u4e00-\u9fa5]/g) || []).length;
    const englishWords = (text.match(/[a-zA-Z]+/g) || []).length;
    return Math.ceil(chineseChars + englishWords * 1.3);
  }
}

interface TokenBudget {
  total: number;
  systemPrompt: number;
  characters: number;
  foreshadows: number;
  previousChapters: number;
  worldFacts: number;
  outputFormat: number;
  reserved: number;
}

interface AllocationContext {
  contentLength: number;
  chapterPosition: number;
  complexity: number;
}

interface AllocatedBudget {
  budget: TokenBudget;
  factors: {
    length: BudgetFactors;
    position: BudgetFactors;
    complexity: BudgetFactors;
  };
}

interface BudgetFactors {
  characters: number;
  foreshadows: number;
  previous: number;
}
```

```typescript
class ContextCompressor {
  compressPreviousChapters(chapters: PreviousChapter[], targetTokens: number): string {
    let result = '';
    let currentTokens = 0;

    for (let i = chapters.length - 1; i >= 0; i--) {
      const chapter = chapters[i];
      const summary = this.summarizeChapter(chapter);
      const tokens = this.estimateTokens(summary);

      if (currentTokens + tokens <= targetTokens) {
        result = summary + '\n' + result;
        currentTokens += tokens;
      } else {
        const keyEvents = this.extractKeyEvents(chapter);
        const keyEventsText = `第${chapter.index}章关键事件：${keyEvents.join('；')}`;
        const keyTokens = this.estimateTokens(keyEventsText);

        if (currentTokens + keyTokens <= targetTokens) {
          result = keyEventsText + '\n' + result;
          currentTokens += keyTokens;
        } else {
          break;
        }
      }
    }

    return result.trim();
  }

  compressCharacters(characters: Character[], targetTokens: number): string {
    const sorted = [...characters].sort((a, b) => {
      const importanceA = this.calculateCharacterImportance(a);
      const importanceB = this.calculateCharacterImportance(b);
      return importanceB - importanceA;
    });

    let result = '';
    let currentTokens = 0;

    for (const char of sorted) {
      const info = this.formatCharacterInfo(char, 'compact');
      const tokens = this.estimateTokens(info);

      if (currentTokens + tokens <= targetTokens) {
        result += info + '\n';
        currentTokens += tokens;
      } else {
        const minimal = `${char.name}：${char.personality.slice(0, 20)}...`;
        const minimalTokens = this.estimateTokens(minimal);

        if (currentTokens + minimalTokens <= targetTokens) {
          result += minimal + '\n';
          currentTokens += minimalTokens;
        }
      }
    }

    return result.trim();
  }

  compressForeshadows(foreshadows: Foreshadowing[], targetTokens: number): string {
    const sorted = [...foreshadows].sort((a, b) => b.importance - a.importance);

    let result = '';
    let currentTokens = 0;

    for (const fs of sorted) {
      const info = `【伏笔】${fs.description.slice(0, 30)}...（重要性: ${fs.importance}）`;
      const tokens = this.estimateTokens(info);

      if (currentTokens + tokens <= targetTokens) {
        result += info + '\n';
        currentTokens += tokens;
      } else {
        break;
      }
    }

    return result.trim();
  }

  private summarizeChapter(chapter: PreviousChapter): string {
    return `第${chapter.index}章《${chapter.title}》：${chapter.summary}`;
  }

  private extractKeyEvents(chapter: PreviousChapter): string[] {
    return chapter.keyEvents?.slice(0, 3) || [];
  }

  private calculateCharacterImportance(char: Character): number {
    let score = 0;
    if (char.role === 'PROTAGONIST') score += 10;
    if (char.role === 'DEUTERAGONIST') score += 8;
    if (char.role === 'ANTAGONIST') score += 7;
    return score;
  }

  private formatCharacterInfo(char: Character, mode: 'full' | 'compact'): string {
    if (mode === 'compact') {
      return `${char.name}（${char.role}）：${char.personality}。动机：${char.motivation.slice(0, 30)}...`;
    }
    return `${char.name}：${JSON.stringify(char)}`;
  }

  private estimateTokens(text: string): number {
    const chineseChars = (text.match(/[\u4e00-\u9fa5]/g) || []).length;
    const englishWords = (text.match(/[a-zA-Z]+/g) || []).length;
    return Math.ceil(chineseChars + englishWords * 1.3);
  }
}

interface PreviousChapter {
  index: number;
  title: string;
  summary: string;
  keyEvents?: string[];
}
```

```typescript
class TopologyManager {
  constructor(private storage: Storage) {}

  async getRelatedNodes(
    projectId: string,
    nodeId: string,
    maxDepth: number = 2
  ): Promise<RelatedNodesResult> {
    const topology = await this.storage.getTopology(projectId);
    const visited = new Set<string>();
    const result: TopologyNode[] = [];

    const traverse = (currentId: string, depth: number) => {
      if (depth > maxDepth || visited.has(currentId)) return;

      visited.add(currentId);
      const node = topology.nodes.find((n) => n.id === currentId);
      if (node) result.push(node);

      const relatedEdges = topology.edges.filter(
        (e) => e.source === currentId || e.target === currentId
      );

      for (const edge of relatedEdges) {
        const nextId = edge.source === currentId ? edge.target : edge.source;
        traverse(nextId, depth + 1);
      }
    };

    traverse(nodeId, 0);
    return { nodes: result, totalCount: result.length };
  }

  async getCharacterGraph(projectId: string): Promise<CharacterGraph> {
    const topology = await this.storage.getTopology(projectId);

    const characterNodes = topology.nodes.filter((n) => n.type === 'CHARACTER');
    const characterEdges = topology.edges.filter(
      (e) =>
        e.type === 'RELATIONSHIP' && (e.source.startsWith('char_') || e.target.startsWith('char_'))
    );

    return { nodes: characterNodes, edges: characterEdges };
  }

  async getPlotTimeline(projectId: string): Promise<PlotTimeline> {
    const topology = await this.storage.getTopology(projectId);
    const outline = await this.storage.getOutline(projectId);

    const events = topology.nodes
      .filter((n) => n.type === 'EVENT')
      .sort((a, b) => (a.data.eventChapter || 0) - (b.data.eventChapter || 0));

    return {
      events: events.map((e) => ({
        chapter: e.data.eventChapter,
        description: e.data.eventDescription,
        impact: e.data.eventImpact,
      })),
      totalChapters: outline.totalChapters,
    };
  }
}
```
