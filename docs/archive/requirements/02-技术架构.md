## 2. 核心模块设计

### 2.1 单一入口设计

```typescript
class MoZhouApp {
  private dialogManager: DialogManager;
  private contextManager: ContextManager;
  private workflowOrchestrator: WorkflowOrchestrator;
  private reviewManager: ReviewManager;

  // 唯一入口方法
  async processUserInput(input: UserInput): Promise<AIResponse> {
    // 1. 意图识别
    const intent = await this.intentRecognizer.recognize(input);

    // 2. 上下文更新
    await this.contextManager.update(input, intent);

    // 3. 路由到编排器
    const result = await this.workflowOrchestrator.execute(intent);

    // 4. 生成响应
    return this.responseGenerator.generate(result);
  }
}
```

### 3.1 核心事件类型

```typescript
interface DomainEvent<T = unknown> {
  type: string;
  payload: T;
  metadata: {
    timestamp: number;
    correlationId: string;
    source: string;
  };
}

type CoreEvents =
  | { type: 'PROJECT_CREATED'; payload: Project }
  | { type: 'NODE_EXECUTION_STARTED'; payload: { nodeId: string; type: WorkflowNodeType } }
  | { type: 'NODE_EXECUTION_COMPLETED'; payload: { nodeId: string; output: string } }
  | { type: 'NODE_APPROVAL_REQUESTED'; payload: { nodeId: string; content: string } }
  | { type: 'NODE_APPROVED'; payload: { nodeId: string; approved: boolean } }
  | { type: 'FEEDBACK_RECEIVED'; payload: UserFeedback }
  | { type: 'CONSISTENCY_ISSUE_FOUND'; payload: ConsistencyIssue }
  | { type: 'FORESHADOW_RESOLVED'; payload: { foreshadowId: string; chapterId: string } };
```

### 3.2 事件总线实现

```typescript
class EventBus {
  private handlers: Map<string, Set<EventHandler<any>>> = new Map();

  subscribe<T>(eventType: string, handler: EventHandler<T>): () => void {
    if (!this.handlers.has(eventType)) {
      this.handlers.set(eventType, new Set());
    }
    this.handlers.get(eventType)!.add(handler);

    return () => {
      this.handlers.get(eventType)?.delete(handler);
    };
  }

  emit<T>(event: DomainEvent<T>): void {
    const handlers = this.handlers.get(event.type);
    if (handlers) {
      handlers.forEach((handler) => {
        try {
          handler(event.payload);
        } catch (error) {
          console.error(`Error handling event ${event.type}:`, error);
        }
      });
    }
  }
}
```

### 4.2 工作流执行器

```typescript
class WorkflowOrchestrator {
  private nodes: Map<string, WorkflowNode> = new Map();
  private currentExecution: WorkflowExecution | null = null;

  async startWorkflow(projectId: string, startNodeId: string): Promise<void> {
    this.currentExecution = {
      projectId,
      currentNodeId: startNodeId,
      status: 'running',
      startedAt: Date.now(),
    };

    await this.executeNode(startNodeId);
  }

  private async executeNode(nodeId: string): Promise<void> {
    const node = this.nodes.get(nodeId);
    if (!node) throw new Error(`Node not found: ${nodeId}`);

    this.eventBus.emit({
      type: 'NODE_EXECUTION_STARTED',
      payload: { nodeId, type: node.type },
      metadata: this.createMetadata(),
    });

    try {
      const result = await this.executeNodeLogic(node);

      this.eventBus.emit({
        type: 'NODE_EXECUTION_COMPLETED',
        payload: { nodeId, output: result },
        metadata: this.createMetadata(),
      });

      const nextNodeId = this.determineNextNode(node, result);
      if (nextNodeId) {
        await this.executeNode(nextNodeId);
      }
    } catch (error) {
      await this.handleNodeError(nodeId, error);
    }
  }

  private async executeNodeLogic(node: WorkflowNode): Promise<any> {
    switch (node.type) {
      case WorkflowNodeType.SETUP_GENERATION:
        return await this.aiEngine.generateSetup(node.config);
      case WorkflowNodeType.OUTLINE_GENERATION:
        return await this.aiEngine.generateOutline(node.config);
      case WorkflowNodeType.CONTENT_GENERATION:
        return await this.aiEngine.generateContent(node.config);
      case WorkflowNodeType.SETUP_REVIEW:
      case WorkflowNodeType.OUTLINE_REVIEW:
      case WorkflowNodeType.CONTENT_REVIEW:
        return await this.waitForUserApproval(node);
      default:
        throw new Error(`Unknown node type: ${node.type}`);
    }
  }

  private determineNextNode(currentNode: WorkflowNode, result: any): string | null {
    for (const transition of currentNode.transitions) {
      if (this.evaluateCondition(transition.condition, result)) {
        return transition.targetNodeId;
      }
    }
    return null;
  }
}
```

## 5. 扩展架构：多 Agent 协作（成长阶段）

在成熟阶段，可引入多 Agent 架构作为可选扩展层，将单一 AI Engine 拆分为多个专业化 Agent：

```
┌─────────────────────────────────────────────────────────────────────┐
│                    Multi-Agent Layer (扩展层)                        │
│  ┌──────────────┐  ┌──────────────┐  ┌──────────────────────────┐  │
│  │   Planner    │  │   Writer     │  │      Reviewer            │  │
│  │   Agent      │  │   Agent      │  │      Agent               │  │
│  └──────────────┘  └──────────────┘  └──────────────────────────┘  │
│  ┌──────────────┐                                                  │
│  │  Optimizer   │                                                  │
│  │   Agent      │                                                  │
│  └──────────────┘                                                  │
└─────────────────────────────────────────────────────────────────────┘
```

### 5.1 Agent 职责

| Agent | 职责 |
|-------|------|
| **Planner** | 负责设定和大纲的规划、调整 |
| **Writer** | 负责章节正文生成 |
| **Reviewer** | 负责内容审核、一致性检查 |
| **Optimizer** | | 负责根据反馈优化 Prompt 和生成策略 |

### 5.2 协调方式

```typescript
class AgentOrchestrator {
  async executeTask(task: AgentTask): Promise<AgentResult> {
    // 1. Planner 分解任务
    const plan = await this.plannerAgent.plan(task);

    // 2. Writer 执行写作
    const draft = await this.writerAgent.write(plan);

    // 3. Reviewer 审核内容
    const review = await this.reviewerAgent.review(draft);

    // 4. 如有问题，Optimizer 提出优化建议
    if (review.issues.length > 0) {
      const optimization = await this.optimizerAgent.optimize(draft, review);
      return this.writerAgent.rewrite(draft, optimization);
    }

    return draft;
  }
}
```
