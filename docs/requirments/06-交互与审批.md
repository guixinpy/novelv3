## 1. 对话状态机

### 1.1 状态定义

```typescript
enum DialogState {
  IDLE = 'idle',
  COLLECTING = 'collecting',
  GENERATING = 'generating',
  PENDING_REVIEW = 'pending_review',
  REVISION = 'revision',
}

interface DialogStateMachine {
  currentState: DialogState;
  context: DialogContext;
}
```

### 1.2 状态转换表

| 当前状态 | 事件 | 目标状态 | 动作 |
|----------|------|----------|------|
| `IDLE` | `USER_INPUT` / `START_PROJECT` | `COLLECTING` | 初始化信息收集 |
| `COLLECTING` | `INFO_COMPLETE` | `GENERATING` | 启动 AI 生成 |
| `COLLECTING` | `MORE_INFO_NEEDED` | `COLLECTING` | 继续收集 |
| `GENERATING` | `GENERATION_COMPLETE` | `PENDING_REVIEW` | 展示审批面板 |
| `GENERATING` | `GENERATION_FAILED` | `IDLE` | 显示错误 |
| `PENDING_REVIEW` | `APPROVED` | `IDLE` | 保存并继续 |
| `PENDING_REVIEW` | `REJECTED` | `IDLE` | 丢弃并通知 |
| `PENDING_REVIEW` | `FIX_REQUESTED` | `REVISION` | 启动修正 |
| `REVISION` | `REVISION_COMPLETE` | `PENDING_REVIEW` | 重新提交审批 |
| `REVISION` | `REVISION_FAILED` | `IDLE` | 显示错误 |

### 1.3 状态管理器

```typescript
class DialogStateManager {
  private state: DialogState = DialogState.IDLE;
  private context: DialogContext = {};

  async transition(event: StateEvent): Promise<void> {
    const transition = this.findTransition(this.state, event.type);
    if (!transition) {
      throw new Error(`Invalid transition from ${this.state} on event ${event.type}`);
    }

    await this.executeExitAction(this.state);
    const oldState = this.state;
    this.state = transition.targetState;
    await this.executeEntryAction(this.state, event);
  }

  private findTransition(
    currentState: DialogState,
    eventType: string
  ): StateTransition | undefined {
    const transitions: Record<DialogState, Record<string, StateTransition>> = {
      [DialogState.IDLE]: {
        USER_INPUT: { targetState: DialogState.COLLECTING },
        START_PROJECT: { targetState: DialogState.COLLECTING },
      },
      [DialogState.COLLECTING]: {
        INFO_COMPLETE: { targetState: DialogState.GENERATING },
        MORE_INFO_NEEDED: { targetState: DialogState.COLLECTING },
      },
      [DialogState.GENERATING]: {
        GENERATION_COMPLETE: { targetState: DialogState.PENDING_REVIEW },
        GENERATION_FAILED: { targetState: DialogState.IDLE },
      },
      [DialogState.PENDING_REVIEW]: {
        APPROVED: { targetState: DialogState.IDLE },
        REJECTED: { targetState: DialogState.IDLE },
        FIX_REQUESTED: { targetState: DialogState.REVISION },
      },
      [DialogState.REVISION]: {
        REVISION_COMPLETE: { targetState: DialogState.PENDING_REVIEW },
        REVISION_FAILED: { targetState: DialogState.IDLE },
      },
    };
    return transitions[currentState]?.[eventType];
  }

  private async executeEntryAction(state: DialogState, event: StateEvent): Promise<void> {
    switch (state) {
      case DialogState.COLLECTING:
        await this.startInformationCollection(event.payload);
        break;
      case DialogState.GENERATING:
        await this.startGeneration(event.payload);
        break;
      case DialogState.PENDING_REVIEW:
        await this.showReviewPanel(event.payload);
        break;
      case DialogState.REVISION:
        await this.startRevision(event.payload);
        break;
    }
  }

  private async executeExitAction(state: DialogState): Promise<void> {
    // 状态退出清理
  }

  getCurrentState(): DialogState {
    return this.state;
  }
}
```

### 2.2 意图识别器

```typescript
class IntentRecognizer {
  private patterns: Map<IntentType, IntentPattern[]> = new Map();

  constructor(private aiService?: AIService) {
    this.initializePatterns();
  }

  async recognize(input: string): Promise<Intent> {
    const normalizedInput = input.toLowerCase().trim();
    let bestMatch: Intent | null = null;
    let highestScore = 0;

    for (const [intentType, patterns] of this.patterns) {
      const score = this.calculateMatchScore(normalizedInput, patterns);
      if (score > highestScore) {
        highestScore = score;
        bestMatch = {
          type: intentType,
          confidence: score,
          entities: this.extractEntities(normalizedInput, intentType),
          rawInput: input,
        };
      }
    }

    if (!bestMatch || bestMatch.confidence < 0.5) {
      return this.aiService
        ? await this.aiAssistedRecognition(input)
        : { type: IntentType.GENERAL_CHAT, confidence: 0.5, entities: {}, rawInput: input };
    }

    return bestMatch;
  }

  private initializePatterns(): void {
    this.patterns.set(IntentType.CREATE_PROJECT, [
      { type: 'keyword', words: ['新建', '创建', '开始', '写', '创作'] },
      { type: 'keyword', words: ['小说', '故事', '书', '作品'] },
    ]);
    this.patterns.set(IntentType.APPROVE, [
      { type: 'keyword', words: ['批准', '通过', '同意', '可以', '没问题', '好'] },
    ]);
    this.patterns.set(IntentType.REJECT, [
      { type: 'keyword', words: ['不通过', '拒绝', '不行', '重', '换'] },
    ]);
    this.patterns.set(IntentType.QUERY_STATUS, [
      { type: 'keyword', words: ['进度', '状态', '写到哪里', '完成'] },
    ]);
    this.patterns.set(IntentType.VIEW_TOPOLOGY, [
      { type: 'keyword', words: ['拓扑', '关系', '角色关系', '情节'] },
    ]);
    this.patterns.set(IntentType.EXPORT, [
      { type: 'keyword', words: ['导出', '下载', '保存', '输出'] },
    ]);
  }

  private calculateMatchScore(input: string, patterns: IntentPattern[]): number {
    let score = 0;
    for (const pattern of patterns) {
      if (pattern.type === 'keyword' && pattern.words) {
        for (const word of pattern.words) {
          if (input.includes(word)) score += 0.3;
        }
      }
    }
    return Math.min(score, 1);
  }

  private extractEntities(input: string, intentType: IntentType): Record<string, unknown> {
    const entities: Record<string, unknown> = {};

    const projectNameMatch = input.match(/["']([^"']+)["']/);
    if (projectNameMatch) entities.projectName = projectNameMatch[1];

    const genreMap: Record<string, string> = {
      玄幻: 'FANTASY', 科幻: 'SCIFI', 言情: 'ROMANCE',
      悬疑: 'MYSTERY', 武侠: 'WUXIA', 仙侠: 'XIANXIA',
    };
    for (const [keyword, genre] of Object.entries(genreMap)) {
      if (input.includes(keyword)) {
        entities.genre = genre;
        break;
      }
    }
    return entities;
  }

  private async aiAssistedRecognition(input: string): Promise<Intent> {
    // 由 AI 服务辅助识别，失败时回退到通用聊天
    return { type: IntentType.GENERAL_CHAT, confidence: 0.5, entities: {}, rawInput: input };
  }
}

interface IntentPattern {
  type: 'keyword' | 'regex';
  words?: string[];
  regex?: RegExp;
}
```

### 3.2 响应生成器

```typescript
class ResponseGenerator {
  generate(intent: Intent, result: unknown): AIResponse {
    switch (intent.type) {
      case IntentType.CREATE_PROJECT:
        return this.generateProjectCreatedResponse(result as Project);
      case IntentType.GENERATE_SETUP:
        return this.generateSetupResponse(result as Setup);
      case IntentType.GENERATE_OUTLINE:
        return this.generateOutlineResponse(result as Outline);
      case IntentType.GENERATE_CONTENT:
        return this.generateContentResponse(result as ChapterContent);
      case IntentType.QUERY_STATUS:
        return this.generateStatusResponse(result as Project);
      default:
        return this.generateGenericResponse(result);
    }
  }

  private generateProjectCreatedResponse(project: Project): AIResponse {
    return {
      type: ResponseType.CARD,
      content: `项目 "${project.name}" 已创建`,
      metadata: { projectId: project.id },
      actions: [
        { label: '开始设计设定', type: 'button', action: 'generate_setup' },
        { label: '查看项目详情', type: 'button', action: 'view_project' },
      ],
    };
  }

  private generateSetupResponse(setup: Setup): AIResponse {
    return {
      type: ResponseType.RICH_CONTENT,
      content: this.formatSetupContent(setup),
      metadata: { requiresReview: true },
      actions: [
        { label: '批准，继续生成大纲', type: 'button', action: 'approve_setup' },
        { label: '需要修改', type: 'button', action: 'request_fix_setup' },
        { label: '重新生成', type: 'button', action: 'regenerate_setup' },
      ],
    };
  }

  private generateOutlineResponse(outline: Outline): AIResponse {
    return {
      type: ResponseType.RICH_CONTENT,
      content: this.formatOutlineContent(outline),
      metadata: { requiresReview: true },
      actions: [
        { label: '批准，开始写作', type: 'button', action: 'approve_outline' },
        { label: '需要修改', type: 'button', action: 'request_fix_outline' },
        { label: '重新生成', type: 'button', action: 'regenerate_outline' },
      ],
    };
  }

  private generateContentResponse(chapter: ChapterContent): AIResponse {
    return {
      type: ResponseType.RICH_CONTENT,
      content: this.formatChapterContent(chapter),
      metadata: { requiresReview: true, chapterIndex: chapter.chapterIndex },
      actions: [
        { label: '批准，继续下一章', type: 'button', action: 'approve_chapter' },
        { label: '需要修改', type: 'button', action: 'request_fix_chapter' },
        { label: '重写', type: 'button', action: 'regenerate_chapter' },
      ],
    };
  }

  private generateStatusResponse(project: Project): AIResponse {
    const progress = Math.round((project.currentWordCount / project.targetWordCount) * 100);
    return {
      type: ResponseType.CARD,
      content: `当前阶段: ${project.currentPhase}\n已完成: ${project.currentWordCount} / ${project.targetWordCount} 字\n进度: ${progress}%`,
    };
  }

  private generateGenericResponse(result: unknown): AIResponse {
    return {
      type: ResponseType.TEXT,
      content: typeof result === 'string' ? result : JSON.stringify(result, null, 2),
    };
  }

  private formatSetupContent(setup: Setup): string {
    // 格式化设定内容
    return JSON.stringify(setup, null, 2);
  }

  private formatOutlineContent(outline: Outline): string {
    return JSON.stringify(outline, null, 2);
  }

  private formatChapterContent(chapter: ChapterContent): string {
    return chapter.content;
  }
}
```

```typescript
class ReviewManager {
  constructor(
    private storage: Storage,
    private eventBus: EventBus,
    private feedbackCollector: FeedbackCollector
  ) {}

  async submitReview(params: {
    projectId: string;
    targetType: 'SETUP' | 'OUTLINE' | 'CHAPTER';
    targetId: string;
    approved: boolean;
    comment?: string;
    correctionSuggestion?: string;
    rating?: number;
  }): Promise<Review> {
    const review: Review = {
      id: generateUUID(),
      projectId: params.projectId,
      targetType: params.targetType,
      targetId: params.targetId,
      status: params.approved
        ? ReviewStatus.APPROVED
        : params.correctionSuggestion
        ? ReviewStatus.FIX_REQUESTED
        : ReviewStatus.REJECTED,
      comment: params.comment,
      correctionSuggestion: params.correctionSuggestion,
      rating: params.rating,
      createdAt: Date.now(),
      reviewedAt: Date.now(),
    };

    await this.storage.saveReview(review);

    // 同步更新目标状态
    await this.updateTargetStatus(params.targetType, params.targetId, review.status);

    // 收集反馈
    await this.feedbackCollector.collectApprovalFeedback({
      projectId: params.projectId,
      nodeId: params.targetId,
      approved: params.approved,
      comment: params.comment,
      rating: params.rating,
    });

    // 发布审批事件
    this.eventBus.emit({
      type: 'NODE_APPROVED',
      payload: { nodeId: params.targetId, approved: params.approved },
      metadata: { timestamp: Date.now(), correlationId: review.id, source: 'ReviewManager' },
    });

    return review;
  }

  async getPendingReviews(projectId: string): Promise<Review[]> {
    return this.storage.getReviewsByProject(projectId, ReviewStatus.PENDING);
  }

  private async updateTargetStatus(
    targetType: string,
    targetId: string,
    status: ReviewStatus
  ): Promise<void> {
    switch (targetType) {
      case 'SETUP':
        await this.storage.updateSetup(targetId, { reviewStatus: status });
        break;
      case 'OUTLINE':
        await this.storage.updateOutline(targetId, { reviewStatus: status });
        break;
      case 'CHAPTER':
        await this.storage.updateChapterContent(targetId, { reviewStatus: status });
        break;
    }
  }
}
```

```typescript
  constructor(private storage: Storage) {}

  async createVersion(
    projectId: string,
    targetType: VersionTargetType,
    targetId: string,
    content: string,
    changeDescription: string,
    createdBy: 'ai_system' | 'user' = 'ai_system'
  ): Promise<Version> {
    const versions = await this.getVersions(projectId, targetType, targetId);
    const newVersionNumber = versions.length > 0 ? Math.max(...versions.map((v) => v.number)) + 1 : 1;

    const version: Version = {
      id: generateUUID(),
      projectId,
      targetType,
      targetId,
      number: newVersionNumber,
      content,
      changeDescription,
      createdAt: Date.now(),
      createdBy,
      metadata: { wordCount: content.length },
    };

    await this.storage.saveVersion(version);
    return version;
  }

  async getVersions(
    projectId: string,
    targetType: VersionTargetType,
    targetId: string
  ): Promise<Version[]> {
    return this.storage.getVersions(projectId, targetType, targetId);
  }

  async getLatestVersion(
    projectId: string,
    targetType: VersionTargetType,
    targetId: string
  ): Promise<Version | null> {
    const versions = await this.getVersions(projectId, targetType, targetId);
    return versions.length > 0 ? versions.sort((a, b) => b.number - a.number)[0] : null;
  }

  async compareVersions(versionId1: string, versionId2: string): Promise<VersionDiff> {
    const v1 = await this.storage.getVersion(versionId1);

      `Rollback to version ${versionNumber}`,
```

## 7. 审批与版本集成

版本管理嵌入写作流程的关键节点：

1. **AI 生成后**：自动创建初始版本（`createdBy: 'ai_system'`）。
2. **用户修正后**：创建修订版本（`createdBy: 'user'`）。
3. **审批面板**：展示当前内容，并提供"查看历史版本"和"对比版本"入口。
4. **回滚操作**：用户可从版本列表选择历史版本进行回滚，回滚本身也记录为新版本。
