## 2. 重试策略

### 2.1 重试配置

```typescript
interface RetryConfig {
  maxAttempts: number;
  baseDelay: number;
  maxDelay: number;
  backoffMultiplier: number;
  retryableErrors: string[];
}

const DEFAULT_RETRY_CONFIG: RetryConfig = {
  maxAttempts: 3,
  baseDelay: 1000,
  maxDelay: 10000,
  backoffMultiplier: 2,
  retryableErrors: ['AI_SERVICE_ERROR', 'NETWORK_ERROR', 'TIMEOUT'],
};
```

### 2.2 重试执行器

```typescript
async function withRetry<T>(
  fn: () => Promise<T>,
  config: RetryConfig = DEFAULT_RETRY_CONFIG
): Promise<T> {
  let lastError: Error | undefined;

  for (let attempt = 1; attempt <= config.maxAttempts; attempt++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error as Error;

      const isRetryable =
        error instanceof AppError &&
        config.retryableErrors.includes(error.code);

      if (!isRetryable || attempt === config.maxAttempts) {
        throw error;
      }

      const delay = Math.min(
        config.baseDelay * Math.pow(config.backoffMultiplier, attempt - 1),
        config.maxDelay
      );
      await sleep(delay);
    }
  }

  throw lastError;
}
```

### 3.2 缓存管理器

```typescript
class CacheManager {
  private caches: Map<string, LRUCache<unknown>> = new Map();

  constructor(config: CacheStrategy = DEFAULT_CACHE_STRATEGY) {
    this.caches.set('projectMetadata', new LRUCache(config.projectMetadata));
    this.caches.set('chapterContent', new LRUCache(config.chapterContent));
    this.caches.set('topology', new LRUCache(config.topology));
    this.caches.set('aiResponse', new LRUCache(config.aiResponse));
  }

  get<T>(cacheName: string, key: string): T | undefined {
    return this.caches.get(cacheName)?.get(key) as T | undefined;
  }

  set<T>(cacheName: string, key: string, value: T): void {
    this.caches.get(cacheName)?.set(key, value);
  }

  invalidate(cacheName: string, key?: string): void {
    const cache = this.caches.get(cacheName);
    if (!cache) return;
    if (key) cache.delete(key);
    else cache.clear();
  }
}

class LRUCache<T> {
  private cache: Map<string, CacheEntry<T>> = new Map();

  constructor(private config: { maxSize: number; ttl: number }) {}

  get(key: string): T | undefined {
    const entry = this.cache.get(key);
    if (!entry) return undefined;
    if (Date.now() - entry.timestamp > this.config.ttl) {
      this.cache.delete(key);
      return undefined;
    }
    // 更新访问时间
    this.cache.delete(key);
    this.cache.set(key, { ...entry, timestamp: Date.now() });
    return entry.value;
  }

  set(key: string, value: T): void {
    if (this.cache.has(key)) {
      this.cache.delete(key);
    } else if (this.cache.size >= this.config.maxSize) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
    this.cache.set(key, { value, timestamp: Date.now() });
  }

  delete(key: string): void {
    this.cache.delete(key);
  }

  clear(): void {
    this.cache.clear();
  }
}

interface CacheEntry<T> {
  value: T;
  timestamp: number;
}

}

  success: boolean;
  data?: unknown;
  error?: string;
}
```

### 4.2 插件管理器

```typescript
class PluginManager {
  private plugins: Map<string, Plugin> = new Map();
  private hooks: Map<string, HookHandler[]> = new Map();

  async register(plugin: Plugin): Promise<void> {
    if (this.plugins.has(plugin.id)) {
      throw new Error(`Plugin ${plugin.id} is already registered`);
    }
    await plugin.initialize();
    this.plugins.set(plugin.id, plugin);
    for (const [hookName, handler] of Object.entries(plugin.hooks)) {
      this.registerHook(hookName, handler);
    }
  }

  async unregister(pluginId: string): Promise<void> {
    const plugin = this.plugins.get(pluginId);
    if (!plugin) return;
    for (const hookName of Object.keys(plugin.hooks)) {
      const handlers = this.hooks.get(hookName) || [];
      this.hooks.set(
        hookName,
        handlers.filter((h) => h.handler !== plugin.hooks[hookName].handler)
      );
    }
    await plugin.destroy();
    this.plugins.delete(pluginId);
  }

  async executeHook(hookName: string, context: HookContext): Promise<HookResult> {
    const handlers = this.hooks.get(hookName) || [];
    const sortedHandlers = [...handlers].sort((a, b) => b.priority - a.priority);
    let currentData = context.data;

    for (const handler of sortedHandlers) {
      try {
        const result = await handler.handler({ ...context, data: currentData });
        if (!result.success) return result;
        currentData = result.data ?? currentData;
      } catch (error) {
        return { success: false, error: (error as Error).message };
      }
    }

    return { success: true, data: currentData };
  }

  private registerHook(hookName: string, handler: HookHandler): void {
    if (!this.hooks.has(hookName)) {
      this.hooks.set(hookName, []);
    }
    this.hooks.get(hookName)!.push(handler);
  }
}
```

interface LogContext {
  projectId?: string;
  userId?: string;
  requestId?: string;

  duration?: number;

interface LogEntry {
  timestamp: string;
  level: 'debug' | 'info' | 'warn' | 'error';
  message: string;
  projectId?: string;
  userId?: string;
  requestId?: string;
  operation?: string;
  duration?: number;
  metadata?: Record<string, unknown>;
}

class StructuredLogger {
  debug(message: string, context?: LogContext): void {
    this.log('debug', message, context);
  }
  info(message: string, context?: LogContext): void {
    this.log('info', message, context);
  }
  warn(message: string, context?: LogContext): void {
    this.log('warn', message, context);
  }
  error(message: string, context?: LogContext): void {
    this.log('error', message, context);
  }

  private log(level: LogEntry['level'], message: string, context?: LogContext): void {
    const entry: LogEntry = {
      timestamp: new Date().toISOString(),
      level,
      message,
      ...context,
    };
    // 实际项目中可扩展为写入文件或发送到日志服务
    const output = level === 'error' ? console.error : console.log;
    output(JSON.stringify(entry));
  }
}

### 5.2 性能监控

```typescript
interface PerformanceMetrics {
  generation: {
    totalRequests: number;
    averageLatency: number;
    errorRate: number;
  };
  cache: {
    hitRate: number;
    missRate: number;
  };
}

class PerformanceMonitor {
  private metrics: PerformanceMetrics = {
    generation: { totalRequests: 0, averageLatency: 0, errorRate: 0 },
    cache: { hitRate: 0, missRate: 0 },
  };

  recordGenerationLatency(latency: number, success: boolean): void {
    const gen = this.metrics.generation;
    gen.totalRequests++;
    gen.averageLatency += (latency - gen.averageLatency) / gen.totalRequests;
    if (!success) {
      gen.errorRate = (gen.errorRate * (gen.totalRequests - 1) + 1) / gen.totalRequests;
    }
  }

  recordCacheAccess(hit: boolean): void {
    const cache = this.metrics.cache;
    const total = cache.hitRate + cache.missRate;
    if (hit) {
      cache.hitRate = (cache.hitRate * total + 1) / (total + 1);
    } else {
      cache.missRate = (cache.missRate * total + 1) / (total + 1);
    }
  }

  getMetrics(): PerformanceMetrics {
    return { ...this.metrics };
  }
}
```
