## 1. 反馈系统架构

### 1.1 渐进式实现策略

| 阶段 | 功能 | 复杂度 | 优先级 |
|------|------|--------|--------|
| **MVP** | 基础反馈收集 + 短期记忆 | 低 | P0 |
| **成长** | 偏好学习 + Prompt 优化 | 中 | P1 |
| **成熟** | 长期记忆 + Few-shot 库 + 用户画像 | 高 | P2 |

### 1.3 反馈收集器

```typescript
class FeedbackCollector {
  constructor(
    private storage: Storage,
    private eventBus: EventBus,
    private aiService?: AIService
  ) {}

  async collectApprovalFeedback(params: {
    projectId: string;
    nodeId: string;
    approved: boolean;
    comment?: string;
    rating?: number;
  }): Promise<UserFeedback> {
    const feedback = this.buildFeedback(params.projectId, params.nodeId, {
      type: params.approved ? FeedbackType.CONTENT_APPROVAL : FeedbackType.CONTENT_REVISION,
      severity: params.approved ? FeedbackSeverity.POSITIVE : FeedbackSeverity.MAJOR,
      content: { userComment: params.comment, rating: params.rating },
    });

    await this.storage.saveFeedback(feedback);
    this.emitFeedbackEvent(feedback);
    return feedback;
  }

  async collectComment(params: {
    projectId: string;
    nodeId: string;
    comment: string;
    originalText?: string;
  }): Promise<UserFeedback> {
    const analysis = this.aiService ? await this.analyzeComment(params.comment) : undefined;

    const feedback = this.buildFeedback(params.projectId, params.nodeId, {
      type: analysis ? this.inferFeedbackType(analysis) : FeedbackType.TEXT_COMMENT,
      severity: analysis?.severity || FeedbackSeverity.SUGGESTION,
      content: { originalText: params.originalText, userComment: params.comment },
      analysisResult: analysis,
    });

    await this.storage.saveFeedback(feedback);
    this.emitFeedbackEvent(feedback);
    return feedback;
  }

  private buildFeedback(
    projectId: string,
    nodeId: string,
    partial: Partial<UserFeedback>
  ): UserFeedback {
    return {
      id: generateUUID(),
      projectId,
      nodeId,
      type: FeedbackType.TEXT_COMMENT,
      severity: FeedbackSeverity.SUGGESTION,
      timestamp: Date.now(),
      content: {},
      context: { promptVersion: '1.0', aiModel: 'default', generationParams: {} },
      processingStatus: 'pending',
      ...partial,
    } as UserFeedback;
  }

  private emitFeedbackEvent(feedback: UserFeedback): void {
    this.eventBus.emit({
      type: 'FEEDBACK_RECEIVED',
      payload: feedback,
      metadata: { timestamp: Date.now(), correlationId: feedback.id, source: 'FeedbackCollector' },
    });
  }

  private async analyzeComment(comment: string): Promise<FeedbackAnalysis> {
    // AI 辅助分析评论分类与情感
    // 若 AI 服务不可用，返回默认中性分析
    return {
      categories: ['other'],
      severity: FeedbackSeverity.SUGGESTION,
      sentiment: 'neutral',
      keywords: [],
      confidence: 0.5,
    };
  }

  private inferFeedbackType(analysis: FeedbackAnalysis): FeedbackType {
    const map: Record<string, FeedbackType> = {
      style: FeedbackType.STYLE_PREFERENCE,
      plot: FeedbackType.PLOT_FEEDBACK,
      character: FeedbackType.CHARACTER_OOC,
      consistency: FeedbackType.CONSISTENCY_CORRECTION,
      pacing: FeedbackType.PACING_FEEDBACK,
    };
    for (const category of analysis.categories) {
      if (map[category]) return map[category];
    }
    return FeedbackType.TEXT_COMMENT;
  }
}
```

### 2.2 偏好学习器

```typescript
class PreferenceLearner {
  constructor(
    private storage: Storage,
    private aiService?: AIService
  ) {}

  async learnFromFeedback(feedback: UserFeedback): Promise<void> {
    const profile = await this.getOrCreateProfile(feedback.projectId);

    switch (feedback.type) {
      case FeedbackType.STYLE_PREFERENCE:
        this.learnStylePreference(profile, feedback);
        break;
      case FeedbackType.PACING_FEEDBACK:
        this.learnPacingPreference(profile, feedback);
        break;
      case FeedbackType.PLOT_FEEDBACK:
        this.learnPlotPreference(profile, feedback);
        break;
      case FeedbackType.CHARACTER_OOC:
        this.learnCharacterPreference(profile, feedback);
        break;
      case FeedbackType.EXPLICIT_RATING:
        await this.learnFromRating(profile, feedback);
        break;
    }

    profile.feedbackCount++;
    profile.confidence = this.calculateConfidence(profile);
    profile.lastUpdated = Date.now();

    await this.storage.savePreferenceProfile(profile);
  }

  private learnStylePreference(profile: UserPreferenceProfile, feedback: UserFeedback): void {
    const comment = feedback.content.userComment || '';
    const keywords: Record<string, { preference: keyof UserPreferenceProfile['stylePreferences']; direction: number }> = {
      太详细: { preference: 'descriptionDensity', direction: -0.2 },
      太简略: { preference: 'descriptionDensity', direction: 0.2 },
      对话太多: { preference: 'dialogueRatio', direction: -0.2 },
      对话太少: { preference: 'dialogueRatio', direction: 0.2 },
      太快: { preference: 'pacingSpeed', direction: -0.2 },
      太慢: { preference: 'pacingSpeed', direction: 0.2 },
      太沉重: { preference: 'darknessLevel', direction: -0.2 },
      太轻松: { preference: 'darknessLevel', direction: 0.2 },
    };

    for (const [keyword, config] of Object.entries(keywords)) {
      if (comment.includes(keyword)) {
        const current = profile.stylePreferences[config.preference];
        profile.stylePreferences[config.preference] = this.clamp(current + config.direction, -1, 1);
      }
    }
  }

  private learnPacingPreference(profile: UserPreferenceProfile, feedback: UserFeedback): void {
    const comment = feedback.content.userComment || '';
    if (comment.includes('太快') || comment.includes('赶') || comment.includes('仓促')) {
      profile.stylePreferences.pacingSpeed -= 0.3;
    } else if (comment.includes('太慢') || comment.includes('拖沓') || comment.includes('冗长')) {
      profile.stylePreferences.pacingSpeed += 0.3;
    }
    profile.stylePreferences.pacingSpeed = this.clamp(profile.stylePreferences.pacingSpeed, -1, 1);
  }

  private learnPlotPreference(profile: UserPreferenceProfile, feedback: UserFeedback): void {
    // 根据情节反馈更新情节偏好
  }

  private learnCharacterPreference(profile: UserPreferenceProfile, feedback: UserFeedback): void {
    // 根据角色反馈更新角色偏好
  }

  private async learnFromRating(
    profile: UserPreferenceProfile,
    feedback: UserFeedback
  ): Promise<void> {
    const rating = feedback.content.rating;
    if (!rating) return;

    if (rating >= 4) {
      for (const key of Object.keys(profile.stylePreferences)) {
        const pref = key as keyof UserPreferenceProfile['stylePreferences'];
        const current = profile.stylePreferences[pref];
        profile.stylePreferences[pref] = this.clamp(current * 1.1, -1, 1);
      }
    } else if (rating <= 2 && this.aiService) {
      // 低评分时由 AI 分析调整方向
    }
  }

  private calculateConfidence(profile: UserPreferenceProfile): number {
    const baseConfidence = Math.min(profile.feedbackCount / 20, 1);
    const preferences = Object.values(profile.stylePreferences);
    const extremeness = preferences.reduce((sum, p) => sum + Math.abs(p), 0) / preferences.length;
    return Math.max(0, baseConfidence - extremeness * 0.2);
  }

  private clamp(value: number, min: number, max: number): number {
    return Math.max(min, Math.min(max, value));
  }

  private async getOrCreateProfile(projectId: string): Promise<UserPreferenceProfile> {
    let profile = await this.storage.getPreferenceProfile(projectId);
    if (!profile) {
      profile = {
        id: `up_${generateUUID()}`,
        projectId,
        stylePreferences: {
          descriptionDensity: 0,
          dialogueRatio: 0,
          pacingSpeed: 0,
          emotionalDepth: 0,
          humorLevel: 0,
          darknessLevel: 0,
        },
        plotPreferences: {
          preferredArchetypes: [],
          dislikedTropes: [],
          preferredPacing: 'steady',
          preferredEnding: 'happy',
        },
        characterPreferences: {
          preferredTraits: [],
          dislikedTraits: [],
          preferredGrowth: 'dramatic',
        },
        confidence: 0,
        lastUpdated: Date.now(),
        feedbackCount: 0,
      };
    }
    return profile;
  }
}
```

```typescript
interface MappingRule {
  id: string;
  name: string;
  condition: {
    feedbackTypes?: FeedbackType[];
    categories?: string[];
    keywords?: string[];
    severity?: FeedbackSeverity[];
  };
  targetSection: 'system_role' | 'style_guidelines' | 'character_context' | 'constraints';
  action: {
    type: 'append' | 'replace' | 'insert' | 'remove';
    template: string;
    priority: number;
  };
  activation: {
    minOccurrences: number;
    timeWindow: number;
    confidenceThreshold: number;
  };
}
```

### 3.2 Prompt 优化器

```typescript
class PromptOptimizer {
  private rules: Map<string, MappingRule> = new Map();
  private feedbackCounter: Map<string, number> = new Map();

  constructor() {
    for (const rule of DEFAULT_MAPPING_RULES) {
      this.rules.set(rule.id, rule);
    }
  }

  async mapFeedbackToPromptChanges(feedback: UserFeedback): Promise<PromptChange[]> {
    const changes: PromptChange[] = [];

    for (const rule of this.rules.values()) {
      if (this.matchesRule(feedback, rule)) {
        const counterKey = `${rule.id}:${feedback.projectId}`;
        const count = (this.feedbackCounter.get(counterKey) || 0) + 1;
        this.feedbackCounter.set(counterKey, count);

        if (count >= rule.activation.minOccurrences) {
          const change = this.createPromptChange(feedback, rule);
          if (change) changes.push(change);
        }
      }
    }

    return changes.sort((a, b) => b.priority - a.priority);
  }

  private matchesRule(feedback: UserFeedback, rule: MappingRule): boolean {
    const { condition } = rule;
    if (condition.feedbackTypes && !condition.feedbackTypes.includes(feedback.type)) return false;
    if (condition.categories && feedback.analysisResult) {
      const hasCategory = condition.categories.some((cat) =>
        feedback.analysisResult!.categories.includes(cat)
      );
      if (!hasCategory) return false;
    }
    if (condition.keywords && feedback.content.userComment) {
      const hasKeyword = condition.keywords.some((kw) =>
        feedback.content.userComment!.includes(kw)
      );
      if (!hasKeyword) return false;
    }
    if (condition.severity && !condition.severity.includes(feedback.severity)) return false;
    return true;
  }

  private createPromptChange(feedback: UserFeedback, rule: MappingRule): PromptChange {
    let content = rule.action.template
      .replace(/{userComment}/g, feedback.content.userComment || '')
      .replace(/{originalText}/g, feedback.content.originalText || '')
      .replace(/{suggestedChange}/g, feedback.content.suggestedChange || '');

    return {
      id: generateUUID(),
      targetSection: rule.targetSection,
      action: rule.action.type,
      content,
      priority: rule.action.priority,
      sourceFeedbackId: feedback.id,
      ruleId: rule.id,
      expiresAt: Date.now() + 7 * 24 * 60 * 60 * 1000,
    };
  }
}

interface PromptChange {
  id: string;
  targetSection: string;
  action: 'append' | 'replace' | 'insert' | 'remove';
  content: string;
  priority: number;
  sourceFeedbackId: string;
  ruleId: string;
  expiresAt: number;
}
```

### 4.2 示例库管理器

```typescript
class FewShotExampleLibrary {
  private examples: Map<string, FewShotExample[]> = new Map();

  addExample(example: FewShotExample): void {
    const category = this.examples.get(example.category) || [];
    category.push(example);
    this.examples.set(example.category, category);
  }

  selectExamples(
    category: ExampleCategory,
    context: ExampleSelectionContext,
    count: number = 2
  ): FewShotExample[] {
    const candidates = this.examples.get(category) || [];
    const scored = candidates.map((example) => ({
      example,
      score: this.calculateMatchScore(example, context),
    }));

    return scored
      .sort((a, b) => b.score - a.score)
      .slice(0, count)
      .map((s) => s.example);
  }

  private calculateMatchScore(example: FewShotExample, context: ExampleSelectionContext): number {
    let score = example.quality * 10 + example.successRate * 20;
    if (context.tags) {
      const matchingTags = example.tags.filter((tag) => context.tags!.includes(tag));
      score += matchingTags.length * 15;
    }
    if (context.style && example.tags.includes(context.style)) {
      score += 20;
    }
    score -= example.usageCount * 2;
    return score;
  }

  updateSuccessRate(exampleId: string, success: boolean): void {
    for (const examples of this.examples.values()) {
      const example = examples.find((e) => e.id === exampleId);
      if (example) {
        example.usageCount++;
        const alpha = 0.3;
        example.successRate = example.successRate * (1 - alpha) + (success ? 1 : 0) * alpha;
        example.updatedAt = Date.now();
        break;
      }
    }
  }
}
```

## 5. 自优化闭环

```
用户输入
    │
    ▼
AI 生成 ──→ 用户审批 ──→ 反馈收集
    ▲                      │
    │                      ▼
    └────── Prompt 优化 ←── 偏好学习
                │
                ▼
          Few-shot 示例更新
```

**闭环说明**：

1. 每次审批产生反馈数据。
2. `FeedbackCollector` 将反馈存入存储并触发事件。
3. `PreferenceLearner` 订阅反馈事件，更新用户偏好画像。
4. `PromptOptimizer` 根据累积反馈生成 Prompt 修正片段。
5. 下一次生成时，`PromptManager` 将有效的 Prompt 修正和偏好参数注入生成上下文。
